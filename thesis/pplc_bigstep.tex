\judgbox{d\pplc d'}{$d$ is substitutes to $d'$ inside the evaluation boundary}

\begin{mathpar}
  \Infer{\pplcl{}Const}{
  }{c\pplc c}
  \and
  \Infer{\pplcl{}Ap}{
    d_1\pplc d_1' \\
    d_2\pplc d_2'
  }{d_1\ d_2\pplc d_1'\ d_2'}
  \and
  \Infer{\pplcl{}Closure}{
    \env\pplc\env' \\
    \env'\vdash d\pplc d'
  }{[\env]d\pplc d'}
\end{mathpar}

\judgbox{\env\pplc\env'}{$\env$ substitutes to $\env'$ (outside the evaluation boundary)}

\begin{mathpar}
  \Infer{\pplcl{}EnvNull}{}{\varnothing\pplc\varnothing}
  \and
  \Infer{\pplcl{}Env}{
    \env\pplc\env' \\
    d\pplc d'
  }{\env,x\leftarrow d\pplc\env',x\leftarrow d'}
\end{mathpar}

\judgbox{\env\vdash d\pplc d'}{$d$ substitutes to $d'$ outside the evaluation boundary}

\begin{mathpar}
  \Infer{\pplclo{}Const}{
  }{c\pplc c}
  \and
  \Infer{\pplclo{}BoundVar}{}{\env,x\leftarrow d\vdash x\pplc d}
  \and
  \Infer{\pplclo{}UnboundVar}{
    x\not\in\env
  }{\env,\vdash x\pplc x}
  \and
  \Infer{\pplclo{}Lam}{
    \env\vdash d\pplc d'
  }{\env\vdash\lambda x.d\pplc\lambda x.d'}
  \and
  \Infer{\pplclo{}Ap}{
    \env\vdash d_1\pplc d_1' \\
    \env\vdash d_2\pplc d_2'
  }{\env\vdash d_1(d_2)\pplc d_1'(d_2')}
  \and
  \Infer{\pplclo{}EHole}{
  }{\env\vdash\hehole^u\pplc[\env]\hehole^u}
  \and
  \Infer{\pplclo{}NEHole}{
    \env\vdash d\pplc d'
  }{\env\vdash\hhole{d}^u\pplc[\env]\hhole{d'}^u}
\end{mathpar}
