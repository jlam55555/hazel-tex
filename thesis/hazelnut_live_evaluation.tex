\judgbox{d\Downarrow d'}{$d$ evaluates to $d'$}

\begin{mathpar}
  \Infer{EFinal}{d\textsf{ final}}{d\Downarrow d}
  \and
  \Infer{EAp}{
    d_1\Downarrow\lambda x:\tau.d_1' \\
    d_2\Downarrow d_2' \\
    [d_2'/x]d_1'\Downarrow d
  }{d_1\ d_2\Downarrow d}
  \and
  \Infer{EApCast}{
    d_1\Downarrow d_1'\langle\tau_1\to\tau_2\Rightarrow\tau_1'\to\tau_2'\rangle \\
    \tau_1\to\tau_2\ne\tau_1'\to\tau_2' \\
    d_2\Downarrow d_2' \\
    d_1'(d_2'\langle\tau_1'\Rightarrow\tau_1\rangle))\langle\tau_2\Rightarrow\tau_2'\rangle\Downarrow d
  }{d_1\ d_2\Downarrow d}
  \and
  \Infer{ECastId}{
    d\Downarrow d'
  }{d\langle\tau\Rightarrow\tau\rangle\Downarrow d'}
  \and
  \Infer{ECastSucceed}{
    d\Downarrow d'
  }{d\langle\tau\Rightarrow\tehole\Rightarrow\tau\rangle\Downarrow d'}
  \and
  \Infer{ECastFail}{
    \tau_1\ne\tau_2 \\
    \tau_1\textsf{ ground} \\
    \tau_2\textsf{ ground} \\
    d\Downarrow d'
  }{d\langle\tau_1\Rightarrow\tehole\Rightarrow\tau_2\rangle\Downarrow d\langle\tau_1\Rightarrow\tehole\not\Rightarrow\tau_2\rangle}
  \and
  \Infer{EGround}{
    \arrmatch{\tau}{\tau'} \\
    d\Downarrow d'
  }{d\langle\tau\Rightarrow\tehole\rangle\Downarrow d'\langle\tau\Rightarrow\tau'\Rightarrow\tehole\rangle}
  \and
  \Infer{EExpand}{
    \arrmatch{\tau}{\tau'} \\
    d\Downarrow d'
  }{d\langle\tehole\Rightarrow\tau\rangle\Downarrow d'\langle\tehole\Rightarrow\tau'\Rightarrow\tau\rangle}
  \and
  \Infer{ENEHole}{
    d\Downarrow d'
  }{\hhole{d}_\env^u\Downarrow\hhole{d'}_\env^u}
\end{mathpar}
